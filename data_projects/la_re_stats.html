Transform data into manageable formats.

# Convert bed/bath char values into separate integers
df <-
  df %>%
  mutate(street_address = paste(df$street_num, df$street_name)) %>%
  separate(bed_bath,
           into = c('bed', 'bath'),
           sep = '/') %>%
  separate(
    bath,
    into = c('full', '3q', 'half', 'q'),
    sep = ',',
    convert = T
  ) %>%
  mutate(bath = full + `3q` * .75 + half * .5 + q * .25) %>% 
  select(-c('full', '3q', 'half', 'q'))

# Remove '$' from prices
df$org_list <- sapply(df$org_list, function(x) {
  gsub('\\$|,', '', x)
})
df$new_list <- sapply(df$new_list, function(x) {
  gsub('\\$|,', '', x)
})

# Convert variables to numeric type
df$org_list <- as.numeric(df$org_list)
df$new_list <- as.numeric(df$new_list)
df$bed <- as.numeric(df$bed)

# Convert dates into date type
df$list_date <- as.Date(df$list_date, format = '%m/%d/%Y')
df$close_date <- as.Date(df$close_date, format = '%m/%d/%Y')

# Select variables
df_closed_reg <- 
  df_closed %>%
  select(built_yr, bed, bath, sq_ft, lot_size, new_list) 

# Filter out rows with missing values
df_closed_reg <- 
  df_closed_reg %>% 
  filter(rowSums(is.na(df_closed_reg)) == 0)

X <- 
  df_closed_reg %>%
  select(built_yr, bed, bath, sq_ft, lot_size)

y <- df_closed_reg %>% 
  select(new_list)

X <- data.matrix(X)
y <- data.matrix(y)

set.seed(123)

cvmodel = cv.glmnet(X, y)

Simplified rough interpretation of the above results:

  • On average, each additional bathroom in a property raises the price $9k
  • On average, each square foot raises the price of a property $23

This linear model doesnt reflect the reality where there are diminishing marginal returns for additional bathrooms and square footage when the number of bathrooms and square footage is already high.
